读完聊天记录，意思是 LLM 的 URL 和 API Key 不要硬编码在 `.env` 里，要在系统里有一个**配置页面**，管理员可以在界面上填写、保存、测试连接 🎯

---

**实现方案：**

配置存数据库 → 后端提供读写接口 → 前端做配置页面 → AI打分时从数据库读配置动态初始化客户端

```python
from app.core.database import Base


class LLMConfig(Base):

    id = Column(Integer, primary_key=True, index=True)

    max_retries = Column(Integer, default=3)            # 最大重试次数
    is_active = Column(Boolean, default=False)          # 是否为当前激活配置
    created_at = Column(DateTime, server_default=func.now())
    updated_at = Column(DateTime, server_default=func.now(), onupdate=func.now())
```

## 第三步：路由 `backend/app/api/routes/llm_config.py`

```python
import time
import logging
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update
from openai import OpenAI

from app.core.database import get_db
from app.models.llm_config import LLMConfig
from app.schemas.llm_config import (
    LLMConfigCreate, LLMConfigUpdate,
    LLMConfigOut, LLMConfigTest, LLMConfigTestResult,
)
from app.api.deps import require_admin

router = APIRouter(prefix="/llm-configs", tags=["LLM配置"])
logger = logging.getLogger(__name__)


def _mask_api_key(key: str) -> str:
    """API Key 打码，只显示前4位和后4位"""
    if len(key) <= 8:
        return "****"
    return key[:4] + "****" + key[-4:]


# ── 获取所有配置 ──────────────────────────────────────────
@router.get("", response_model=list[LLMConfigOut])
async def list_configs(
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    result = await db.execute(select(LLMConfig).order_by(LLMConfig.id))
    configs = result.scalars().all()
    # 对 api_key 打码
    for c in configs:
        c.api_key = _mask_api_key(c.api_key)
    return configs


# ── 获取当前激活配置（不打码，内部用） ───────────────────
@router.get("/active", response_model=LLMConfigOut)
async def get_active_config(
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    result = await db.execute(
        select(LLMConfig).where(LLMConfig.is_active == True)
    )
    config = result.scalar_one_or_none()
    if not config:
        raise HTTPException(status_code=404, detail="尚未设置激活的LLM配置")
    config.api_key = _mask_api_key(config.api_key)
    return config


# ── 新建配置 ──────────────────────────────────────────────
@router.post("", response_model=LLMConfigOut)
async def create_config(
    data: LLMConfigCreate,
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    config = LLMConfig(**data.model_dump())
    db.add(config)
    await db.commit()
    await db.refresh(config)
    config.api_key = _mask_api_key(config.api_key)
    return config


# ── 更新配置 ──────────────────────────────────────────────
@router.put("/{config_id}", response_model=LLMConfigOut)
async def update_config(
    config_id: int,
    data: LLMConfigUpdate,
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    result = await db.execute(select(LLMConfig).where(LLMConfig.id == config_id))
    config = result.scalar_one_or_none()
    if not config:
        raise HTTPException(status_code=404, detail="配置不存在")

    update_data = data.model_dump(exclude_none=True)
    for k, v in update_data.items():
        setattr(config, k, v)

    await db.commit()
    await db.refresh(config)
    config.api_key = _mask_api_key(config.api_key)
    return config


# ── 删除配置 ──────────────────────────────────────────────
@router.delete("/{config_id}")
async def delete_config(
    config_id: int,
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    result = await db.execute(select(LLMConfig).where(LLMConfig.id == config_id))
    config = result.scalar_one_or_none()
    if not config:
        raise HTTPException(status_code=404, detail="配置不存在")
    if config.is_active:
        raise HTTPException(status_code=400, detail="不能删除当前激活的配置，请先切换到其他配置")
    await db.delete(config)
    await db.commit()
    return {"message": "已删除"}


# ── 激活某个配置 ──────────────────────────────────────────
@router.post("/{config_id}/activate")
async def activate_config(
    config_id: int,
    db: AsyncSession = Depends(get_db),
    _=Depends(require_admin),
):
    # 先把所有配置设为非激活
    await db.execute(update(LLMConfig).values(is_active=False))

    # 激活指定配置
    result = await db.execute(select(LLMConfig).where(LLMConfig.id == config_id))
    config = result.scalar_one_or_none()
    if not config:
        raise HTTPException(status_code=404, detail="配置不存在")

    config.is_active = True
    await db.commit()
    return {"message": f"已激活配置：{config.name}"}


# ── 测试连接（不保存，直接用传入的参数测试） ─────────────
@router.post("/test", response_model=LLMConfigTestResult)
async def test_connection(
    data: LLMConfigTest,
    _=Depends(require_admin),
):
    start = time.time()
    try:
        client = OpenAI(
            base_url=data.base_url,
            api_key=data.api_key,
            timeout=15,
        )
        response = client.chat.completions.create(
            model=data.model_name,
            messages=[{"role": "user", "content": "reply with the word OK only"}],
            max_tokens=10,
            temperature=0,
        )
        latency_ms = int((time.time() - start) * 1000)
        reply = response.choices[0].message.content or ""
        return LLMConfigTestResult(
            success=True,
            message=f"连接成功，模型响应：{reply.strip()[:50]}",
            latency_ms=latency_ms,
        )
    except Exception as e:
        return LLMConfigTestResult(
            success=False,
            message=f"连接失败：{str(e)[:200]}",
        )
```

## 第四步：修改 `ai_service.py`，从数据库动态读取配置

````python
import json
import re
import logging
from pathlib import Path
from typing import Optional
from dataclasses import dataclass

from openai import OpenAI
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

logger = logging.getLogger(__name__)


@dataclass
class ScoreResult:
    skills: float        # 满分 25
    experience: float    # 满分 25
    education: float     # 满分 20
    potential: float     # 满分 15
    stability: float     # 满分 15
    total: float         # 满分 100
    summary: str
    strengths: list[str]
    weaknesses: list[str]


DEFAULT_SCORE = ScoreResult(
    skills=0, experience=0, education=0,
    potential=0, stability=0, total=0,
    summary="AI打分失败，请人工审核",
    strengths=[],
    weaknesses=["AI自动打分未能完成，需人工评估"],
)


class AIService:

    async def _get_client(self, db: AsyncSession) -> tuple[OpenAI, str, int, int]:
        """从数据库读取激活的LLM配置，返回 (client, model_name, timeout, max_retries)"""
        from app.models.llm_config import LLMConfig
        result = await db.execute(
            select(LLMConfig).where(LLMConfig.is_active == True)
        )
        config = result.scalar_one_or_none()

        if not config:
            raise RuntimeError("未找到激活的LLM配置，请在系统设置中配置并激活一个LLM")

        client = OpenAI(
            base_url=config.base_url,
            api_key=config.api_key,
            timeout=config.timeout,
        )
        return client, config.model_name, config.timeout, config.max_retries

    # ── 简历文本提取 ──────────────────────────────────────
    async def parse_resume_text(self, file_path: str) -> str:
        path = Path(file_path)
        if not path.exists():
            raise FileNotFoundError(f"简历文件不存在: {file_path}")
        suffix = path.suffix.lower()
        try:
            if suffix == ".pdf":
                return self._extract_pdf(path)
            elif suffix in (".docx", ".doc"):
                return self._extract_docx(path)
            elif suffix == ".txt":
                return path.read_text(encoding="utf-8", errors="ignore")
            else:
                raise ValueError(f"不支持的文件格式: {suffix}")
        except Exception as e:
            logger.error(f"简历文本提取失败: {e}")
            raise

    def _extract_pdf(self, path: Path) -> str:
        import pdfplumber
        texts = []
        with pdfplumber.open(path) as pdf:
            for page in pdf.pages:
                text = page.extract_text()
                if text:
                    texts.append(text)
        return "\n".join(texts)

    def _extract_docx(self, path: Path) -> str:
        from docx import Document
        doc = Document(str(path))
        return "\n".join(p.text for p in doc.paragraphs if p.text.strip())

    def _strip_thinking(self, text: str) -> str:
        return re.sub(r"<think>.*?</think>", "", text, flags=re.DOTALL).strip()

    def _extract_json(self, text: str) -> dict:
        text = self._strip_thinking(text)
        try:
            return json.loads(text)
        except json.JSONDecodeError:
            pass
        match = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", text, re.DOTALL)
        if match:
            try:
                return json.loads(match.group(1))
            except json.JSONDecodeError:
                pass
        match = re.search(r"\{.*\}", text, re.DOTALL)
        if match:
            try:
                return json.loads(match.group())
            except json.JSONDecodeError:
                pass
        raise ValueError(f"无法从模型输出中提取JSON，原始输出：\n{text[:500]}")

    # ── 核心打分方法（新增 db 参数） ─────────────────────
    async def score_resume(
        self,
        db: AsyncSession,
        resume_text: str,
        job_title: str,
        job_requirements: str,
        required_education: str,
        required_experience_years: int,
        required_skills: list[str],
    ) -> ScoreResult:
        client, model_name, timeout, max_retries = await self._get_client(db)

        prompt = self._build_scoring_prompt(
            resume_text=resume_text,
            job_title=job_title,
            job_requirements=job_requirements,
            required_education=required_education,
            required_experience_years=required_experience_years,
            required_skills=required_skills,
        )

        last_error: Optional[Exception] = None

        for attempt in range(1, max_retries + 1):
            try:
                logger.info(f"AI打分第 {attempt} 次尝试，岗位：{job_title}")
                response = client.chat.completions.create(
                    model=model_name,
                    messages=[
                        {
                            "role": "system",
                            "content": (
                                "你是一名专业的HR招聘评估专家。"
                                "你的任务是根据岗位要求对候选人简历进行客观评分。"
                                "你必须严格按照要求的JSON格式输出，不要输出任何其他内容。"
                            ),
                        },
                        {"role": "user", "content": prompt},
                    ],
                    temperature=0.1,
                    max_tokens=1500,
                )
                raw_output = response.choices[0].message.content or ""
                logger.debug(f"模型原始输出（前300字）：{raw_output[:300]}")
                data = self._extract_json(raw_output)
                return self._parse_score_result(data)
            except Exception as e:
                last_error = e
                logger.warning(f"第 {attempt} 次打分失败：{e}")

        raise RuntimeError(f"AI打分在 {max_retries} 次重试后仍失败：{last_error}")

    def _build_scoring_prompt(
        self,
        resume_text: str,
        job_title: str,
        job_requirements: str,
        required_education: str,
        required_experience_years: int,
        required_skills: list[str],
    ) -> str:
        edu_map = {
            "any": "不限", "junior_college": "大专",
            "bachelor": "本科", "master": "硕士", "phd": "博士",
        }
        edu_label = edu_map.get(required_education, required_education)
        skills_str = "、".join(required_skills) if required_skills else "无特殊要求"
        resume_truncated = resume_text[:4000]

        return f"""请对以下候选人简历进行评分。

## 岗位信息
- 岗位名称：{job_title}
- 学历要求：{edu_label}
- 工作年限要求：{required_experience_years}年以上
- 技能要求：{skills_str}
- 岗位要求详情：
{job_requirements}

## 候选人简历
{resume_truncated}

## 评分标准（总分100分）

| 维度 | 满分 | 说明 |
|------|------|------|
| skills（技能匹配） | 25分 | 候选人掌握的技术/专业技能与岗位要求的匹配度 |
| experience（经验匹配） | 25分 | 工作年限、项目经验、行业经验与岗位要求的契合度 |
| education（学历匹配） | 20分 | 候选人学历层次与岗位学历要求的匹配程度 |
| potential（发展潜力） | 15分 | 候选人的学习能力、成长轨迹、未来发展空间 |
| stability（稳定性） | 15分 | 工作稳定性，跳槽频率，平均在职时长 |

## 输出要求
只输出以下JSON格式，不要有任何其他文字：

{{
  "skills": <0-25的数字>,
  "experience": <0-25的数字>,
  "education": <0-20的数字>,
  "potential": <0-15的数字>,
  "stability": <0-15的数字>,
  "total": <以上5项之和>,
  "summary": "<100字以内的中文总结评语>",
  "strengths": ["优势1", "优势2", "优势3"],
  "weaknesses": ["不足1", "不足2"]
}}"""

    def _parse_score_result(self, data: dict) -> ScoreResult:
        def clamp(val, max_val):
            try:
                return max(0.0, min(float(val), float(max_val)))
            except (TypeError, ValueError):
                return 0.0

        skills     = clamp(data.get("skills", 0), 25)
        experience = clamp(data.get("experience", 0), 25)
        education  = clamp(data.get("education", 0), 20)
        potential  = clamp(data.get("potential", 0), 15)
        stability  = clamp(data.get("stability", 0), 15)
        total      = skills + experience + education + potential + stability

        strengths  = data.get("strengths", [])
        weaknesses = data.get("weaknesses", [])
        if not isinstance(strengths, list):
            strengths = [str(strengths)]
        if not isinstance(weaknesses, list):
            weaknesses = [str(weaknesses)]

        return ScoreResult(
            skills=skills, experience=experience, education=education,
            potential=potential, stability=stability,
            total=round(total, 1),
            summary=str(data.get("summary", ""))[:500],
            strengths=strengths[:5],
            weaknesses=weaknesses[:5],
        )


ai_service = AIService()
````

## 第五步：前端配置页面 `frontend/src/pages/LLMConfig.tsx`

```typescript
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import {
  Table, Button, Modal, Form, Input, InputNumber,
  Tag, Space, message, Popconfirm, Alert,
} from 'antd'
import {
  PlusOutlined, CheckCircleOutlined,
  ApiOutlined, DeleteOutlined, EditOutlined,
} from '@ant-design/icons'
import { useState } from 'react'
import http from '@/services/http'
import type { LLMConfig } from '@/types/llm'

export default function LLMConfigPage() {
  const queryClient = useQueryClient()
  const [modalOpen, setModalOpen] = useState(false)
  const [editingId, setEditingId] = useState<number | null>(null)
  const [testResult, setTestResult] = useState<{ success: boolean; message: string; latency_ms?: number } | null>(null)
  const [testing, setTesting] = useState(false)
  const [form] = Form.useForm()

  const { data: configs = [], isLoading } = useQuery({
    queryKey: ['llm-configs'],
    queryFn: async () => {
      const res = await http.get<LLMConfig[]>('/llm-configs')
      return res.data
    },
  })

  const createMutation = useMutation({
    mutationFn: async (data: Partial<LLMConfig>) => {
      const res = await http.post<LLMConfig>('/llm-configs', data)
      return res.data
    },
    onSuccess: () => {
      message.success('配置已保存')
      queryClient.invalidateQueries({ queryKey: ['llm-configs'] })
      setModalOpen(false)
      form.resetFields()
      setTestResult(null)
    },
    onError: (err: any) => message.error(err.response?.data?.detail ?? '保存失败'),
  })

  const updateMutation = useMutation({
    mutationFn: async ({ id, data }: { id: number; data: Partial<LLMConfig> }) => {
      const res = await http.put<LLMConfig>(`/llm-configs/${id}`, data)
      return res.data
    },
    onSuccess: () => {
      message.success('配置已更新')
      queryClient.invalidateQueries({ queryKey: ['llm-configs'] })
      setModalOpen(false)
      form.resetFields()
      setEditingId(null)
      setTestResult(null)
    },
    onError: (err: any) => message.error(err.response?.data?.detail ?? '更新失败'),
  })

  const deleteMutation = useMutation({
    mutationFn: async (id: number) => {
      await http.delete(`/llm-configs/${id}`)
    },
    onSuccess: () => {
      message.success('已删除')
      queryClient.invalidateQueries({ queryKey: ['llm-configs'] })
    },
    onError: (err: any) => message.error(err.response?.data?.detail ?? '删除失败'),
  })

  const activateMutation = useMutation({
    mutationFn: async (id: number) => {
      await http.post(`/llm-configs/${id}/activate`)
    },
    onSuccess: () => {
      message.success('已切换激活配置')
      queryClient.invalidateQueries({ queryKey: ['llm-configs'] })
    },
  })

  const handleTest = async () => {
    const values = form.getFieldsValue()
    if (!values.base_url || !values.model_name) {
      message.warning('请先填写 API 地址和模型名称')
      return
    }
    setTesting(true)
    setTestResult(null)
    try {
      const res = await http.post('/llm-configs/test', {
        base_url: values.base_url,
        api_key: values.api_key || 'lm-studio',
        model_name: values.model_name,
      })
      setTestResult(res.data)
    } catch (err: any) {
      setTestResult({ success: false, message: err.response?.data?.detail ?? '请求失败' })
    } finally {
      setTesting(false)
    }
  }

  const handleEdit = (record: LLMConfig) => {
    setEditingId(record.id)
    form.setFieldsValue({
      name: record.name,
      base_url: record.base_url,
      api_key: '',   // 不回填真实key，让用户重新输入
      model_name: record.model_name,
      timeout: record.timeout,
      max_retries: record.max_retries,
    })
    setTestResult(null)
    setModalOpen(true)
  }

  const handleSubmit = (values: any) => {
    if (editingId) {
      // 如果 api_key 为空，不更新 key
      const data = { ...values }
      if (!data.api_key) delete data.api_key
      updateMutation.mutate({ id: editingId, data })
    } else {
      createMutation.mutate(values)
    }
  }

  const columns = [
    { title: 'ID', dataIndex: 'id', width: 60 },
    { title: '配置名称', dataIndex: 'name' },
    { title: 'API 地址', dataIndex: 'base_url' },
    { title: '模型名称', dataIndex: 'model_name' },
    { title: '超时(秒)', dataIndex: 'timeout', width: 90 },
    { title: '重试次数', dataIndex: 'max_retries', width: 90 },
    {
      title: '状态',
      dataIndex: 'is_active',
      width: 90,
      render: (v: boolean) =>
        v ? <Tag color="success" icon={<CheckCircleOutlined />}>激活中</Tag>
          : <Tag color="default">未激活</Tag>,
    },
    {
      title: '操作',
      render: (_: unknown, record: LLMConfig) => (
        <Space>
          <Button
            size="small"
            icon={<EditOutlined />}
            onClick={() => handleEdit(record)}
          >
            编辑
          </Button>
          {!record.is_active && (
            <Button
              size="small"
              type="primary"
              icon={<CheckCircleOutlined />}
              onClick={() => activateMutation.mutate(record.id)}
              loading={activateMutation.isPending}
            >
              激活
            </Button>
          )}
          {!record.is_active && (
            <Popconfirm
              title="确认删除这个配置？"
              onConfirm={() => deleteMutation.mutate(record.id)}
            >
              <Button size="small" danger icon={<DeleteOutlined />}>
                删除
              </Button>
            </Popconfirm>
          )}
        </Space>
      ),
    },
  ]

  return (
    <div>
      <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: 16 }}>
        <div>
```

**整体流程打通后是这样的 🎯：**

```
管理员 → 系统设置 → LLM配置页面
  → 填写 URL / API Key / 模型名
  → 点"测试连接"验证是否通
  → 点"激活"设为当前使用的配置

候选人投递简历
  → 触发 AI 打分
  → ai_service 从数据库读激活配置
  → 动态创建 OpenAI client
  → 打分结果写回数据库
```
