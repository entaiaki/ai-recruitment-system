```python
from fastapi import Depends, HTTPException, status
from app.models.user import User, UserRole
from app.core.security import get_current_active_user


def require_roles(*roles: UserRole):
    """
    用法：current_user: User = Depends(require_roles(UserRole.admin, UserRole.hr))
    只允许指定角色访问，其余一律 403。
    """
    async def checker(
        current_user: User = Depends(get_current_active_user),
    ) -> User:
        if current_user.role not in roles:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"需要权限：{[r.value for r in roles]}，当前角色：{current_user.role.value}",
            )
        return current_user
    return checker


# 预定义常用组合，直接 Depends 不用传参
require_admin      = require_roles(UserRole.admin)
require_hr_or_above = require_roles(UserRole.admin, UserRole.hr)
require_any_staff  = require_roles(UserRole.admin, UserRole.hr, UserRole.dept_manager)
```

## P1-C：抽取 `ApplicationService`

新建 `backend/app/services/application_service.py`：

```python
import json
import logging
from datetime import datetime

from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

from app.models.application import Application, ApplicationStatus, VALID_TRANSITIONS
from app.models.job import Job
from app.models.candidate import Candidate
from app.models.resume import Resume
from app.models.audit_log import AuditLog

logger = logging.getLogger(__name__)


class NotFoundError(Exception):
    """资源不存在"""


class InvalidTransitionError(Exception):
    """非法状态流转"""
    def __init__(self, from_status: ApplicationStatus,
                 to_status: ApplicationStatus,
                 allowed: list[ApplicationStatus]):
        self.from_status = from_status
        self.to_status = to_status
        self.allowed = allowed
        super().__init__(
            f"不允许从 {from_status.value} 流转到 {to_status.value}，"
            f"当前允许：{[s.value for s in allowed]}"
        )


class ApplicationService:
    def __init__(self, db: AsyncSession, operator_id: int):
        self.db = db
        self.operator_id = operator_id

    # ── 提交投递 ──────────────────────────────────────────
    async def submit(
        self,
        job_id: int,
        candidate_id: int,
        resume_id: int,
        cover_letter: str | None = None,
    ) -> Application:
        job       = await self._get_or_404(Job, job_id, "岗位")
        candidate = await self._get_or_404(Candidate, candidate_id, "候选人")
        resume    = await self._get_or_404(Resume, resume_id, "简历")

        # 同一候选人不能重复投递同一岗位（未关闭状态）
        existing = await self.db.execute(
            select(Application).where(
                Application.job_id == job_id,
                Application.candidate_id == candidate_id,
                Application.status.not_in([
                    ApplicationStatus.rejected,
                ]),
            )
        )
        if existing.scalar_one_or_none():
            raise ValueError("该候选人已投递此岗位，请勿重复提交")

        app = Application(
            job_id=job_id,
            candidate_id=candidate_id,
            resume_id=resume_id,
            cover_letter=cover_letter,
            status=ApplicationStatus.submitted,
        )
        self.db.add(app)
        await self.db.flush()   # 拿到 app.id

        self._audit(
            action="create_application",
            resource="application",
            resource_id=str(app.id),
            detail={"job_id": job_id, "candidate_id": candidate_id},
        )

        await self.db.refresh(app)
        logger.info(f"投递创建成功 app_id={app.id} job={job.title}")
        return app

    # ── 状态流转 ──────────────────────────────────────────
    async def transition(
        self,
        app_id: int,
        new_status: ApplicationStatus,
        notes: str | None = None,
        interview_date: datetime | None = None,
    ) -> Application:
        app = await self._get_or_404(Application, app_id, "投递记录")

        allowed = VALID_TRANSITIONS.get(app.status, [])
        if new_status not in allowed:
            raise InvalidTransitionError(app.status, new_status, allowed)

        if new_status == ApplicationStatus.rejected and not notes:
            raise ValueError("拒绝操作必须填写拒绝原因")

        old_status = app.status
        app.status = new_status

        # 根据新状态写入附加信息
        if new_status == ApplicationStatus.hr_review:
            app.reviewed_by_hr = self.operator_id
            if notes:
                app.hr_notes = notes

        elif new_status == ApplicationStatus.dept_review:
            app.reviewed_by_dept = self.operator_id
            if notes:
                app.dept_notes = notes

        elif new_status == ApplicationStatus.interview:
            app.interview_date = interview_date

        elif new_status == ApplicationStatus.rejected:
            app.reject_reason = notes

        self._audit(
            action="update_status",
            resource="application",
            resource_id=str(app_id),
            detail={
                "from": old_status.value,
                "to": new_status.value,
                "notes": notes,
            },
        )

        await self.db.flush()
        await self.db.refresh(app)
        logger.info(f"状态流转 app_id={app_id} {old_status.value} → {new_status.value}")
        return app

    # ── 内部工具 ──────────────────────────────────────────
    async def _get_or_404(self, model, id_: int, label: str):
        obj = await self.db.get(model, id_)
        if not obj:
            raise NotFoundError(f"{label}不存在（id={id_}）")
        return obj

    def _audit(
        self,
        action: str,
        resource: str,
        resource_id: str = "",
        detail: dict | None = None,
    ) -> None:
        """写审计日志，统一入口"""
        self.db.add(AuditLog(
            user_id=self.operator_id,
            action=action,
            resource=resource,
            resource_id=resource_id,
            detail=json.dumps(detail, ensure_ascii=False) if detail else None,
        ))
```

```python
import logging
from datetime import datetime
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func

from app.core.database import get_db
from app.api.deps import (
    require_hr_or_above, require_any_staff, require_admin,
)
from app.core.security import get_current_active_user
from app.models.user import User, UserRole
from app.models.application import Application, ApplicationStatus
from app.schemas.schemas import (
    ApplicationCreate, ApplicationOut,
    ApplicationStatusUpdate, PaginatedResponse,
)
from app.services.application_service import (
    ApplicationService, NotFoundError, InvalidTransitionError,
)

router = APIRouter(prefix="/api/applications", tags=["投递管理"])
logger = logging.getLogger(__name__)


# ── 提交投递 ──────────────────────────────────────────────
@router.post("/", response_model=ApplicationOut, status_code=201)
async def create_application(
    data: ApplicationCreate,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_hr_or_above),
):
    try:
        svc = ApplicationService(db, current_user.id)
        return await svc.submit(
            job_id=data.job_id,
            candidate_id=data.candidate_id,
            resume_id=data.resume_id,
            cover_letter=data.cover_letter,
        )
    except NotFoundError as e:
        raise HTTPException(404, detail=str(e))
    except ValueError as e:
        raise HTTPException(422, detail=str(e))


# ── 更新状态 ──────────────────────────────────────────────
@router.patch("/{app_id}/status", response_model=ApplicationOut)
async def update_status(
    app_id: int,
    update: ApplicationStatusUpdate,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_active_user),
):
    try:
        svc = ApplicationService(db, current_user.id)
        return await svc.transition(
            app_id=app_id,
            new_status=update.new_status,
            notes=update.notes,
            interview_date=update.interview_date,
        )
    except NotFoundError as e:
        raise HTTPException(404, detail=str(e))
    except InvalidTransitionError as e:
        raise HTTPException(422, detail=str(e))
    except ValueError as e:
        raise HTTPException(422, detail=str(e))


# ── 查询列表 ──────────────────────────────────────────────
@router.get("/", response_model=PaginatedResponse)
async def list_applications(
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100),
    status: ApplicationStatus | None = None,
    job_id: int | None = None,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_any_staff),
):
    query = select(Application)

    if status:
        query = query.where(Application.status == status)
    if job_id:
        query = query.where(Application.job_id == job_id)

    # 部门经理只看自己部门
    if current_user.role == UserRole.dept_manager:
        query = query.where(Application.department == current_user.department)

    total = await db.scalar(select(func.count()).select_from(query.subquery()))
    items = await db.execute(
        query.offset((page - 1) * page_size).limit(page_size)
    )

    return {
        "items": items.scalars().all(),
        "total": total,
        "page": page,
        "page_size": page_size,
    }


# ── 查询详情 ──────────────────────────────────────────────
@router.get("/{app_id}", response_model=ApplicationOut)
async def get_application(
    app_id: int,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(require_any_staff),
):
    app = await db.get(Application, app_id)
    if not app:
        raise HTTPException(404, detail="投递记录不存在")
    return app
```

## 改动二：删重复的文件解析，改用 `utils/file_parser`

````python
import json
import re
import logging
from dataclasses import dataclass, field
from typing import Optional

from openai import OpenAI

logger = logging.getLogger(__name__)


# ── 评分结果数据结构 ──────────────────────────────────────
@dataclass
class ScoreResult:
    skills:         float = 0
    experience:     float = 0
    education:      float = 0
    potential:      float = 0
    stability:      float = 0
    total:          float = 0
    summary:        str = ""
    strengths:      list = field(default_factory=list)
    weaknesses:     list = field(default_factory=list)

SCORE_FALLBACK = ScoreResult(

SCORING_SYSTEM_PROMPT = """你是一名专业的 HR 招聘评估专家。

SCORING_USER_TEMPLATE = """请对以下候选人简历进行评分。

## 输出格式（只输出 JSON，不要其他文字）

  "skills":         <0-25>,
  "experience":     <0-25>,
  "education":      <0-20>,
  "potential":      <0-15>,
  "stability":      <0-15>,
  "total":          <以上五项之和>,
  "summary":        "<100 字以内中文总结>",
  "strengths":      ["优势1", "优势2", "优势3"],
  "weaknesses":     ["不足1", "不足2"],
  "recommendation": "<录用 / 待定 / 不推荐>"

EDU_LABEL_MAP = {
    "any":           "不限",
    "junior_college": "大专",
    "bachelor":      "本科",
    "master":        "硕士",

# ── 主服务类 ─────────────────────────────────────────────
class AIService:

    # ── 从数据库读取激活的 LLM 配置 ──────────────────────
    async def _get_client(
        self, db: AsyncSession
    ) -> tuple[OpenAI, str, int, int]:
        """返回 (client, model_name, timeout, max_retries)"""
        from app.models.llm_config import LLMConfig

        result = await db.execute(
            select(LLMConfig).where(LLMConfig.is_active == True)
        )
        config = result.scalar_one_or_none()
        if not config:
            raise RuntimeError("未找到激活的 LLM 配置，请在系统设置中配置并激活")

        client = OpenAI(
            base_url=config.base_url,
            api_key=config.api_key,
            timeout=config.timeout,
        )
        return client, config.model_name, config.timeout, config.max_retries

    # ── 核心打分入口 ──────────────────────────────────────
    async def score_resume(
        self,
        db: AsyncSession,
        resume_text: str,
        job_title: str,
        job_requirements: str,
        required_education: str,
        required_experience_years: int,
        required_skills: list[str],
    ) -> ScoreResult:
        client, model_name, timeout, max_retries = await self._get_client(db)
        prompt = self._build_prompt(
            resume_text=resume_text,
            job_title=job_title,
            job_requirements=job_requirements,
            required_education=required_education,
            required_experience_years=required_experience_years,
            required_skills=required_skills,
        )
        raw = await self._call_with_retry(client, model_name, prompt, max_retries)
        return self._parse(raw)

    # ── 构建 Prompt ───────────────────────────────────────
    def _build_prompt(
        self,
        resume_text: str,
        job_title: str,
        job_requirements: str,
        required_education: str,
        required_experience_years: int,
        required_skills: list[str],
    ) -> str:
        return SCORING_USER_TEMPLATE.format(
            job_title=job_title,
            education=EDU_LABEL_MAP.get(required_education, required_education),
            experience_years=required_experience_years,
            skills="、".join(required_skills) if required_skills else "无特殊要求",
            job_requirements=job_requirements,
            resume_text=resume_text[:4000],   # 防止超出 context window
        )

    # ── 带重试的 API 调用 ─────────────────────────────────
    async def _call_with_retry(
        self,
        client: OpenAI,
        model_name: str,
        prompt: str,
        max_retries: int,
    ) -> str:
        last_error: Optional[Exception] = None

        for attempt in range(1, max_retries + 1):
            try:
                logger.info(f"AI 打分第 {attempt} 次调用")
                response = client.chat.completions.create(
                    model=model_name,
                    messages=[
                        {"role": "system", "content": SCORING_SYSTEM_PROMPT},
                        {"role": "user",   "content": prompt},
                    ],
                    temperature=0.1,
                    max_tokens=1500,
                )
                return response.choices[0].message.content or ""
            except Exception as e:
                last_error = e
                logger.warning(f"第 {attempt} 次调用失败：{e}")

        raise RuntimeError(f"AI 打分在 {max_retries} 次重试后仍失败：{last_error}")

    # ── 解析模型输出 ──────────────────────────────────────
    def _parse(self, raw: str) -> ScoreResult:
        data = self._extract_json(raw)
        return ScoreResult(
            skills=     self._clamp(data.get("skills"),     25),
            experience= self._clamp(data.get("experience"), 25),
            education=  self._clamp(data.get("education"),  20),
            potential=  self._clamp(data.get("potential"),  15),
            stability=  self._clamp(data.get("stability"),  15),
            total=      round(sum([
                self._clamp(data.get("skills"),     25),
                self._clamp(data.get("experience"), 25),
                self._clamp(data.get("education"),  20),
                self._clamp(data.get("potential"),  15),
                self._clamp(data.get("stability"),  15),
            ]), 1),
            summary=        str(data.get("summary", ""))[:500],
            strengths=      self._to_list(data.get("strengths"))[:5],
            weaknesses=     self._to_list(data.get("weaknesses"))[:5],
            recommendation= str(data.get("recommendation", "待评估")),
        )

    # ── 工具方法 ──────────────────────────────────────────
    def _extract_json(self, text: str) -> dict:
        """从模型输出中提取 JSON，兼容 markdown 代码块和 think 标签"""
        text = re.sub(r"<think>.*?</think>", "", text, flags=re.DOTALL).strip()

        # 直接解析
        try:
            return json.loads(text)
        except json.JSONDecodeError:
            pass

        # 从 markdown 代码块提取
        match = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", text, re.DOTALL)
        if match:
            try:
                return json.loads(match.group(1))
            except json.JSONDecodeError:
                pass

        # 从任意位置提取第一个 JSON 对象
        match = re.search(r"\{.*\}", text, re.DOTALL)
        if match:
            try:
                return json.loads(match.group())
            except json.JSONDecodeError:
                pass

        raise ValueError(f"无法从模型输出中提取 JSON，原始输出前 200 字：\n{text[:200]}")

    @staticmethod
    def _clamp(value, max_val: float) -> float:
        try:
            return max(0.0, min(float(value), float(max_val)))
        except (TypeError, ValueError):
            return 0.0

    @staticmethod
    def _to_list(value) -> list:
        if isinstance(value, list):
            return value
        if value is None:
            return []
        return [str(value)]


ai_service = AIService()
````
