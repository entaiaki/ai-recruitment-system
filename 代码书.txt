from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from sqlalchemy.orm import DeclarativeBase
from app.core.config import settings


class Base(DeclarativeBase):
    pass


engine = create_async_engine(
    settings.DATABASE_URL,
    echo=settings.DEBUG,
    pool_pre_ping=True,
    pool_size=10,
    max_overflow=20,
)

AsyncSessionLocal = async_sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False,
    autocommit=False,
    autoflush=False,
)


async def get_db() -> AsyncSession:
    """FastAPI dependency: yields a database session."""
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
        finally:
            await session.close()


async def init_db():
    """Create all tables on startup."""
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.create_all)

from datetime import datetime, timedelta, timezone
from typing import Optional, Any
from jose import JWTError, jwt
from passlib.context import CryptContext
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer
from sqlalchemy.ext.asyncio import AsyncSession
from app.core.config import settings
from app.core.database import get_db

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")
oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/api/auth/login")


def verify_password(plain_password: str, hashed_password: str) -> bool:
    return pwd_context.verify(plain_password, hashed_password)


def get_password_hash(password: str) -> str:
    return pwd_context.hash(password)


def create_access_token(subject: Any, expires_delta: Optional[timedelta] = None) -> str:
    expire = datetime.now(timezone.utc) + (
        expires_delta or timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
    )
    payload = {"sub": str(subject), "exp": expire}
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)


def decode_token(token: str) -> Optional[str]:
    try:
        payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
        return payload.get("sub")
    except JWTError:
        return None


async def get_current_user(
    token: str = Depends(oauth2_scheme),
    db: AsyncSession = Depends(get_db),
):
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    user_id = decode_token(token)
    if user_id is None:
        raise credentials_exception

    from app.models.user import User
    from sqlalchemy import select

    result = await db.execute(select(User).where(User.id == int(user_id)))
    user = result.scalar_one_or_none()
    if user is None:
        raise credentials_exception
    return user


async def get_current_active_user(current_user=Depends(get_current_user)):
    if not current_user.is_active:
        raise HTTPException(status_code=400, detail="Inactive user")
    return current_user

from fastapi import APIRouter, Depends, HTTPException, status
from fastapi.security import OAuth2PasswordRequestForm
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.core.database import get_db
from app.core.security import verify_password, get_password_hash, create_access_token
from app.models.user import User
from app.schemas.user import UserCreate, UserOut, Token

router = APIRouter(prefix="/auth", tags=["auth"])


@router.post("/register", response_model=UserOut, status_code=status.HTTP_201_CREATED)
async def register(user_in: UserCreate, db: AsyncSession = Depends(get_db)):
    # Check duplicate email
    result = await db.execute(select(User).where(User.email == user_in.email))
    if result.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Email already registered")

    user = User(
        email=user_in.email,
        hashed_password=get_password_hash(user_in.password),
        full_name=user_in.full_name,
        role=user_in.role,
    )
    db.add(user)
    await db.flush()
    await db.refresh(user)
    return user


@router.post("/login", response_model=Token)
async def login(
    form_data: OAuth2PasswordRequestForm = Depends(),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(User).where(User.email == form_data.username))
    user = result.scalar_one_or_none()

    if not user or not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect email or password",
            headers={"WWW-Authenticate": "Bearer"},
        )
    if not user.is_active:
        raise HTTPException(status_code=400, detail="Inactive user")

    access_token = create_access_token(subject=user.id)
    return {"access_token": access_token, "token_type": "bearer"}
```

from fastapi import APIRouter, Depends, HTTPException, status, Query
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from typing import List, Optional
from app.core.database import get_db
from app.core.security import get_current_active_user
from app.models.job import Job, JobStatus
from app.models.user import User, UserRole
from app.schemas.job import JobCreate, JobUpdate, JobOut

router = APIRouter(prefix="/jobs", tags=["jobs"])


@router.get("/", response_model=List[JobOut])
async def list_jobs(
    skip: int = 0,
    limit: int = Query(default=20, le=100),
    keyword: Optional[str] = None,
    db: AsyncSession = Depends(get_db),
):
    query = select(Job).where(Job.status == JobStatus.active)
    if keyword:
        query = query.where(Job.title.ilike(f"%{keyword}%"))
    query = query.offset(skip).limit(limit)
    result = await db.execute(query)
    return result.scalars().all()


@router.get("/{job_id}", response_model=JobOut)
async def get_job(job_id: int, db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(Job).where(Job.id == job_id))
    job = result.scalar_one_or_none()
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")
    return job


@router.post("/", response_model=JobOut, status_code=status.HTTP_201_CREATED)
async def create_job(
    job_in: JobCreate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    if current_user.role not in [UserRole.employer, UserRole.admin]:
        raise HTTPException(status_code=403, detail="Only employers can post jobs")

    job = Job(**job_in.model_dump(), employer_id=current_user.id)
    db.add(job)
    await db.flush()
    await db.refresh(job)
    return job


@router.put("/{job_id}", response_model=JobOut)
async def update_job(
    job_id: int,
    job_in: JobUpdate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(Job).where(Job.id == job_id))
    job = result.scalar_one_or_none()
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")
    if job.employer_id != current_user.id and current_user.role != UserRole.admin:
        raise HTTPException(status_code=403, detail="Not authorized")

    for field, value in job_in.model_dump(exclude_unset=True).items():
        setattr(job, field, value)
    await db.flush()
    await db.refresh(job)
    return job


@router.delete("/{job_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_job(
    job_id: int,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(Job).where(Job.id == job_id))
    job = result.scalar_one_or_none()
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")
    if job.employer_id != current_user.id and current_user.role != UserRole.admin:
        raise HTTPException(status_code=403, detail="Not authorized")
    await db.delete(job)

@router.post("/", response_model=ResumeOut, status_code=status.HTTP_201_CREATED)
async def upload_resume(
    title: str = Form(...),
    file: UploadFile = File(...),
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    # Validate file extension
    ext = os.path.splitext(file.filename)[1].lower()
    if ext not in ALLOWED_EXTENSIONS:
        raise HTTPException(
            status_code=400,
            detail=f"File type not allowed. Supported: {ALLOWED_EXTENSIONS}",
        )

    # Validate file size
    content = await file.read()
    if len(content) > settings.MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail="File too large (max 10MB)")

    # Save file to disk
    os.makedirs(settings.UPLOAD_DIR, exist_ok=True)
    unique_name = f"{uuid.uuid4().hex}{ext}"
    file_path = os.path.join(settings.UPLOAD_DIR, unique_name)
    with open(file_path, "wb") as f:
        f.write(content)

    # Parse text content from the file
    from app.utils.file_parser import parse_resume_file
    parsed_content = parse_resume_file(file_path, ext)

    resume = Resume(
        user_id=current_user.id,
        title=title,
        file_path=file_path,
        file_name=file.filename,
        parsed_content=parsed_content,
    )
    db.add(resume)
    await db.flush()
    await db.refresh(resume)
    return resume

@router.post("/", response_model=ApplicationOut, status_code=status.HTTP_201_CREATED)
async def apply_for_job(
    app_in: ApplicationCreate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    # Must be a job seeker
    if current_user.role != UserRole.job_seeker:
        raise HTTPException(status_code=403, detail="Only job seekers can apply")

    # Check job exists and is active
    job_result = await db.execute(select(Job).where(Job.id == app_in.job_id))
    job = job_result.scalar_one_or_none()
    if not job or job.status.value != "active":
        raise HTTPException(status_code=404, detail="Job not found or closed")

    # Check resume belongs to current user
    resume_result = await db.execute(select(Resume).where(Resume.id == app_in.resume_id))
    resume = resume_result.scalar_one_or_none()
    if not resume or resume.user_id != current_user.id:
        raise HTTPException(status_code=403, detail="Invalid resume")

    # Prevent duplicate applications
    dup = await db.execute(
        select(Application).where(
            Application.job_id == app_in.job_id,
            Application.applicant_id == current_user.id,
        )
    )
    if dup.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Already applied to this job")

    application = Application(
        job_id=app_in.job_id,
        applicant_id=current_user.id,
        resume_id=app_in.resume_id,
        cover_letter=app_in.cover_letter,
    )
    db.add(application)
    await db.flush()
    await db.refresh(application)

    # Trigger AI scoring in background (non-blocking)
    from app.services.ai_service import score_application_background
    import asyncio
    asyncio.create_task(
        score_application_background(application.id, job, resume)
    )

    return application

import json
import logging
from typing import Optional
from openai import AsyncOpenAI
from app.core.config import settings

logger = logging.getLogger(__name__)

client = AsyncOpenAI(
    api_key=settings.OPENAI_API_KEY,
    base_url=settings.OPENAI_BASE_URL,
)

SCORE_PROMPT_TEMPLATE = """
You are an expert HR recruiter and talent assessment specialist.

Analyze the following resume against the job requirements and provide a structured evaluation.

## Job Information
Title: {job_title}
Company: {company}
Requirements:
{requirements}

## Resume Content
{resume_content}

## Instructions
Please respond with a JSON object in exactly this format:
{{
  "score": <integer 0-100>,
  "summary": "<2-3 sentence overall assessment>",
  "strengths": ["<strength 1>", "<strength 2>", "<strength 3>"],
  "weaknesses": ["<weakness 1>", "<weakness 2>"],
  "recommendation": "<hire | consider | reject>"
}}

Be objective, concise, and base your assessment purely on the match between the resume and job requirements.
"""


async def score_resume(
    job_title: str,
    company: str,
    requirements: str,
    resume_content: str,
) -> dict:
    """
    Call the AI model to score a resume against a job posting.
    Returns a dict with score, summary, strengths, weaknesses, recommendation.
    """
    prompt = SCORE_PROMPT_TEMPLATE.format(
        job_title=job_title,
        company=company,
        requirements=requirements,
        resume_content=resume_content[:4000],  # Trim to avoid token overflow
    )

    try:
        response = await client.chat.completions.create(
            model=settings.AI_MODEL,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.3,
            max_tokens=800,
            response_format={"type": "json_object"},
        )
        raw = response.choices[0].message.content
        return json.loads(raw)
    except json.JSONDecodeError as e:
        logger.error(f"AI response JSON parse error: {e}")
        return _fallback_score()
    except Exception as e:
        logger.error(f"AI scoring failed: {e}")
        return _fallback_score()


def _fallback_score() -> dict:
    """Return a neutral score when AI call fails."""
    return {
        "score": 50,
        "summary": "AI scoring unavailable. Manual review required.",
        "strengths": [],
        "weaknesses": [],
        "recommendation": "consider",
    }


async def score_application_background(
    application_id: int,
    job,
    resume,
):
    """
    Background task: score an application and write results back to DB.
    Runs after the HTTP response is already returned to the client.
    """
    from app.core.database import AsyncSessionLocal
    from app.models.application import Application
    from sqlalchemy import select

    resume_content = resume.parsed_content or resume.title

    result = await score_resume(
        job_title=job.title,
        company=job.company,
        requirements=job.requirements,
        resume_content=resume_content,
    )

    async with AsyncSessionLocal() as db:
        try:
            query = await db.execute(
                select(Application).where(Application.id == application_id)
            )
            application = query.scalar_one_or_none()
            if application:
                application.ai_score = float(result.get("score", 50))
                application.ai_analysis = result.get("summary", "")
                application.ai_strengths = json.dumps(result.get("strengths", []))
                application.ai_weaknesses = json.dumps(result.get("weaknesses", []))
                await db.commit()
                logger.info(f"AI scored application {application_id}: {application.ai_score}")
        except Exception as e:
            await db.rollback()
            logger.error(f"Failed to save AI score for application {application_id}: {e}")

import logging
from typing import Optional

logger = logging.getLogger(__name__)


def parse_resume_file(file_path: str, ext: str) -> Optional[str]:
    """
    Extract plain text from uploaded resume files.
    Supports: .pdf, .docx, .doc
    """
    try:
        if ext == ".pdf":
            return _parse_pdf(file_path)
        elif ext in (".docx", ".doc"):
            return _parse_docx(file_path)
        else:
            return None
    except Exception as e:
        logger.error(f"Failed to parse resume {file_path}: {e}")
        return None


def _parse_pdf(file_path: str) -> str:
    import PyPDF2
    text_parts = []
    with open(file_path, "rb") as f:
        reader = PyPDF2.PdfReader(f)
        for page in reader.pages:
            page_text = page.extract_text()
            if page_text:
                text_parts.append(page_text)
    return "\n".join(text_parts)


def _parse_docx(file_path: str) -> str:
    from docx import Document
    doc = Document(file_path)
    paragraphs = [p.text for p in doc.paragraphs if p.text.strip()]
    return "\n".join(paragraphs)
```

from datetime import datetime, timezone
from typing import Optional
from sqlalchemy import String, Text, Integer, DateTime, ForeignKey, Enum as SAEnum
from sqlalchemy.orm import Mapped, mapped_column, relationship
import enum
from app.core.database import Base


class JobStatus(str, enum.Enum):
    draft = "draft"       # 草稿
    active = "active"     # 招聘中
    paused = "paused"     # 暂停
    closed = "closed"     # 已关闭


class EducationLevel(str, enum.Enum):
    any = "any"
    junior_college = "junior_college"   # 大专
    bachelor = "bachelor"               # 本科
    master = "master"                   # 硕士
    phd = "phd"                         # 博士


class Job(Base):
    __tablename__ = "jobs"

    id: Mapped[int] = mapped_column(primary_key=True, index=True)
    # 创建人（HR）
    created_by: Mapped[int] = mapped_column(ForeignKey("users.id"), nullable=False)
    # 归属部门负责人
    dept_leader_id: Mapped[Optional[int]] = mapped_column(ForeignKey("users.id"))

    title: Mapped[str] = mapped_column(String(200), nullable=False)
    department: Mapped[str] = mapped_column(String(100), nullable=False)
    description: Mapped[str] = mapped_column(Text, nullable=False)
    requirements: Mapped[str] = mapped_column(Text, nullable=False)

    # 结构化要求字段（供 AI 打分用）
    required_education: Mapped[EducationLevel] = mapped_column(
        SAEnum(EducationLevel), default=EducationLevel.bachelor
    )
    required_experience_years: Mapped[int] = mapped_column(Integer, default=0)
    required_skills: Mapped[Optional[str]] = mapped_column(Text)  # JSON 数组字符串
    headcount: Mapped[int] = mapped_column(Integer, default=1)    # 招聘人数

    location: Mapped[Optional[str]] = mapped_column(String(100))
    salary_min: Mapped[Optional[int]] = mapped_column(Integer)
    salary_max: Mapped[Optional[int]] = mapped_column(Integer)
    status: Mapped[JobStatus] = mapped_column(SAEnum(JobStatus), default=JobStatus.active)

    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),
        default=lambda: datetime.now(timezone.utc)
    )
    updated_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True),

class AIScoreResult:
    """5维度打分结果，对齐任务书 D 模块"""
    def __init__(self, data: dict):
        self.education_match: float = float(data.get("education_match", 0))   # 0-20
        self.experience_match: float = float(data.get("experience_match", 0)) # 0-25
        self.skill_match: float = float(data.get("skill_match", 0))           # 0-30
        self.stability: float = float(data.get("stability", 0))               # 0-15
        self.overall_impression: float = float(data.get("overall_impression", 0))  # 0-10
        self.total: float = (
            self.education_match + self.experience_match +
            self.skill_match + self.stability + self.overall_impression
        )
        self.summary: str = data.get("summary", "")
        self.strengths: list[str] = data.get("strengths", [])
        self.weaknesses: list[str] = data.get("weaknesses", [])


class AIService:
    def __init__(self):
        self.client = AsyncOpenAI(
            api_key=settings.OPENAI_API_KEY,
            base_url=settings.OPENAI_BASE_URL,
        )

    async def score_resume(
        self,
        resume_text: str,
        job_title: str,
        job_requirements: str,
        required_education: str,
        required_experience_years: int,
        required_skills: list[str],
    ) -> AIScoreResult:
        """
        调用大模型对简历进行 5 维度打分。
        严格要求返回 JSON，不接受自由文本。
        """
        prompt = f"""
你是一位专业的企业 HR 和招聘专家。请根据以下岗位要求，对候选人简历进行结构化评分。

## 岗位信息
- 职位：{job_title}
- 学历要求：{required_education}
- 工作年限要求：{required_experience_years} 年
- 技能要求：{', '.join(required_skills)}
- 岗位职责与要求：
{job_requirements}

## 候选人简历
{resume_text}

## 评分规则（总分 100 分）
请严格按照以下 5 个维度打分，每个维度分数不得超过上限：

| 维度 | 字段名 | 满分 | 评分标准 |
|------|--------|------|----------|
| 学历匹配 | education_match | 20 | 完全符合20，差一级扣8分 |
| 经验匹配 | experience_match | 25 | 年限+行业背景综合评估 |
| 技能匹配 | skill_match | 30 | 按技能覆盖率比例给分 |
| 稳定性 | stability | 15 | 平均在职时长，跳槽频繁扣分 |
| 整体印象 | overall_impression | 10 | 简历质量、表达清晰度 |

## 输出格式
只输出以下 JSON，不要有任何额外文字：
{{
  "education_match": <0-20的数字>,
  "experience_match": <0-25的数字>,
  "skill_match": <0-30的数字>,
  "stability": <0-15的数字>,
  "overall_impression": <0-10的数字>,
  "summary": "<100字以内的中文总结>",
  "strengths": ["优势1", "优势2", "优势3"],
  "weaknesses": ["不足1", "不足2"]
}}
"""
        response = await self.client.chat.completions.create(
            model=settings.AI_MODEL,
            messages=[{"role": "user", "content": prompt}],
            temperature=0.2,          # 低温度保证输出稳定
            response_format={"type": "json_object"},
        )

        raw = response.choices[0].message.content
        try:
            data = json.loads(raw)
        except json.JSONDecodeError as e:
            logger.error(f"AI 返回非法 JSON: {raw}")
            raise ValueError(f"AI 返回格式错误: {e}")

        return AIScoreResult(data)

### `backend/app/utils/file_parser.py`

```python
import asyncio
import os
from pathlib import Path


async def extract_text(file_path: str) -> str:
    """
    异步提取文件文本内容，支持 PDF / DOCX / TXT。
    使用 asyncio.to_thread 避免阻塞事件循环。
    """
    return await asyncio.to_thread(_extract_sync, file_path)


def _extract_sync(file_path: str) -> str:
    suffix = Path(file_path).suffix.lower()

    if suffix == ".pdf":
        return _extract_pdf(file_path)
    elif suffix == ".docx":
        return _extract_docx(file_path)
    elif suffix == ".txt":
        with open(file_path, "r", encoding="utf-8", errors="ignore") as f:
            return f.read()
    else:
        raise ValueError(f"不支持的文件类型: {suffix}")


def _extract_pdf(file_path: str) -> str:
    import PyPDF2
    text_parts = []
    with open(file_path, "rb") as f:
        reader = PyPDF2.PdfReader(f)
        for page in reader.pages:
            text = page.extract_text()
            if text:
                text_parts.append(text)
    return "\n".join(text_parts)


def _extract_docx(file_path: str) -> str:
    from docx import Document
    doc = Document(file_path)
    return "\n".join(para.text for para in doc.paragraphs if para.text.strip())

        query = query.where(Application.ai_total_score >= min_score)
    query = query.order_by(Application.ai_total_score.desc().nulls_last())
    query = query.offset(skip).limit(limit)

    result = await db.execute(query)
    return result.scalars().all()


@router.get("/{app_id}", response_model=ApplicationOut)
async def get_application(
    app_id: int,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(Application).where(Application.id == app_id))
    application = result.scalar_one_or_none()
    if not application:
        raise HTTPException(status_code=404, detail="Application not found")
    return application


@router.post("/{app_id}/retry-scoring", response_model=ApplicationOut)
async def retry_scoring(
    app_id: int,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    """对打分失败的申请重新触发 AI 打分"""
    if current_user.role not in [UserRole.hr, UserRole.admin]:
        raise HTTPException(status_code=403, detail="Only HR can retry scoring")

    result = await db.execute(select(Application).where(Application.id == app_id))
    application = result.scalar_one_or_none()
    if not application:
        raise HTTPException(status_code=404, detail="Application not found")

    if application.status != ApplicationStatus.scoring_failed:
        raise HTTPException(
            status_code=400,
            detail=f"Can only retry scoring_failed applications, current: {application.status}"
        )

    # 重置状态，后台重新打分
    application.status = ApplicationStatus.submitted
    application.scoring_error = None
    await db.flush()
    await db.refresh(application)

    import asyncio
    asyncio.create_task(_trigger_ai_scoring(application.id))
    return application

@router.post("/register", response_model=UserOut, status_code=status.HTTP_201_CREATED)
async def register(user_in: UserCreate, db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(User).where(User.email == user_in.email))
    if result.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Email already registered")

    user = User(
        email=user_in.email,
        hashed_password=get_password_hash(user_in.password),
        full_name=user_in.full_name,
        role=user_in.role,
        department=user_in.department,
    )
    db.add(user)
    await db.flush()
    await db.refresh(user)
    return user


@router.post("/login", response_model=Token)
async def login(
    form_data: OAuth2PasswordRequestForm = Depends(),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(User).where(User.email == form_data.username))
    user = result.scalar_one_or_none()

    if not user or not verify_password(form_data.password, user.hashed_password):
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Incorrect email or password",
            headers={"WWW-Authenticate": "Bearer"},
        )
    if not user.is_active:
        raise HTTPException(status_code=400, detail="Inactive user")

    access_token = create_access_token(subject=user.id)
    return {"access_token": access_token, "token_type": "bearer"}

## 第四个：`backend/app/api/routes/applications.py`（完整版）

```python
import json
import asyncio
from fastapi import APIRouter, Depends, HTTPException, status, Query
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from typing import List, Optional
from app.core.database import get_db
from app.core.security import get_current_active_user
from app.models.application import Application, ApplicationStatus, VALID_TRANSITIONS
from app.models.job import Job
from app.models.resume import Resume
from app.models.user import User, UserRole
from app.schemas.application import ApplicationCreate, ApplicationStatusUpdate, ApplicationOut
from app.services.ai_service import ai_service

router = APIRouter(prefix="/applications", tags=["applications"])


@router.post("/", response_model=ApplicationOut, status_code=status.HTTP_201_CREATED)
async def create_application(
    app_in: ApplicationCreate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    if current_user.role not in [UserRole.hr, UserRole.admin]:
        raise HTTPException(status_code=403, detail="Only HR can submit applications")

    job_res = await db.execute(select(Job).where(Job.id == app_in.job_id))
    job = job_res.scalar_one_or_none()
    if not job:
        raise HTTPException(status_code=404, detail="Job not found")

    resume_res = await db.execute(select(Resume).where(Resume.id == app_in.resume_id))
    resume = resume_res.scalar_one_or_none()
    if not resume:
        raise HTTPException(status_code=404, detail="Resume not found")

    application = Application(
        job_id=app_in.job_id,
        candidate_id=resume.candidate_id,
        resume_id=app_in.resume_id,
        cover_letter=app_in.cover_letter,
        status=ApplicationStatus.submitted,
    )
    db.add(application)
    await db.flush()
    await db.refresh(application)

    # 后台触发 AI 打分，不阻塞响应
    asyncio.create_task(_trigger_ai_scoring(application.id))

    return application


async def _trigger_ai_scoring(application_id: int):
    """后台任务：AI 打分流程"""
    from app.core.database import AsyncSessionLocal

    async with AsyncSessionLocal() as db:
        try:
            result = await db.execute(
                select(Application).where(Application.id == application_id)
            )
            application = result.scalar_one_or_none()
            if not application:
                return

            application.status = ApplicationStatus.ai_scoring
            await db.flush()

            resume_result = await db.execute(
                select(Resume).where(Resume.id == application.resume_id)
            )
            resume = resume_result.scalar_one()

            job_result = await db.execute(
                select(Job).where(Job.id == application.job_id)
            )
            job = job_result.scalar_one()

            resume_text = resume.parsed_content or ""
            if not resume_text and resume.file_path:
                resume_text = await ai_service.parse_resume_text(resume.file_path)
                resume.parsed_content = resume_text

            required_skills = json.loads(job.required_skills or "[]")

            score = await ai_service.score_resume(
                resume_text=resume_text,
                job_title=job.title,
                job_requirements=job.requirements,
                required_education=job.required_education.value,
                required_experience_years=job.required_experience_years,
                required_skills=required_skills,
            )

            application.score_education = score.education_match
            application.score_experience = score.experience_match
            application.score_skill = score.skill_match
            application.score_stability = score.stability
            application.score_impression = score.overall_impression
            application.ai_total_score = score.total
            application.ai_summary = score.summary
            application.ai_strengths = json.dumps(score.strengths, ensure_ascii=False)
            application.ai_weaknesses = json.dumps(score.weaknesses, ensure_ascii=False)
            application.status = ApplicationStatus.ai_scored

            await db.commit()

        except Exception as e:
            try:
                application.status = ApplicationStatus.scoring_failed
                application.scoring_error = str(e)
                await db.commit()
            except Exception:
                pass


@router.put("/{app_id}/status", response_model=ApplicationOut)
async def update_status(
    app_id: int,
    update: ApplicationStatusUpdate,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(Application).where(Application.id == app_id))
    application = result.scalar_one_or_none()
    if not application:
        raise HTTPException(status_code=404, detail="Application not found")

    current_status = application.status
    new_status = update.new_status
    allowed = VALID_TRANSITIONS.get(current_status, [])

    if new_status not in allowed:
        raise HTTPException(
            status_code=400,
            detail=f"Invalid transition: {current_status} -> {new_status}. Allowed: {[s.value for s in allowed]}"
        )

    # 角色权限校验
    if new_status == ApplicationStatus.dept_review:
        if current_user.role not in [UserRole.hr, UserRole.admin]:
            raise HTTPException(status_code=403, detail="Only HR can move to dept_review")

    if new_status in [ApplicationStatus.interview, ApplicationStatus.rejected]:
        if current_user.role not in [UserRole.dept_leader, UserRole.admin, UserRole.hr]:
            raise HTTPException(status_code=403, detail="Not authorized for this transition")

    if new_status in [ApplicationStatus.offered, ApplicationStatus.hired]:
        if current_user.role not in [UserRole.hr, UserRole.admin]:
            raise HTTPException(status_code=403, detail="Only HR can issue offers")

    # 记录操作人
    if current_user.role == UserRole.hr:
        application.reviewed_by_hr = current_user.id
        if update.notes:
            application.hr_notes = update.notes
    elif current_user.role == UserRole.dept_leader:
        application.reviewed_by_dept = current_user.id
        if update.notes:
            application.dept_notes = update.notes

    application.status = new_status
    await db.flush()
    await db.refresh(application)
    return application


@router.get("/", response_model=List[ApplicationOut])
async def list_applications(
    job_id: Optional[int] = None,
    status: Optional[ApplicationStatus] = None,
    skip: int = 0,
    limit: int = Query(default=20, le=100),
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    query = select(Application)
    if job_id:
        query = query.where(Application.job_id == job_id)
    if status:
        query = query.where(Application.status == status)
    # analyst 只读，其他角色正常访问
    query = query.offset(skip).limit(limit).order_by(Application.applied_at.desc())
    result = await db.execute(query)
    return result.scalars().all()


@router.get("/{app_id}", response_model=ApplicationOut)
async def get_application(
    app_id: int,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(select(Application).where(Application.id == app_id))
    application = result.scalar_one_or_none()
    if not application:
        raise HTTPException(status_code=404, detail="Application not found")
    return application


@router.post("/{app_id}/retry-scoring", response_model=ApplicationOut)
async def retry_scoring(
    app_id: int,
    current_user: User = Depends(get_current_active_user),
    db: AsyncSession = Depends(get_db),
):
    """对打分失败的申请重新触发 AI 打分"""
    if current_user.role not in [UserRole.hr, UserRole.admin]:
        raise HTTPException(status_code=403, detail="Only HR can retry scoring")

    result = await db.execute(select(Application).where(Application.id == app_id))
    application = result.scalar_one_or_none()
    if not application:
        raise HTTPException(status_code=404, detail="Application not found")

    if application.status != ApplicationStatus.scoring_failed:
        raise HTTPException(
            status_code=400,
            detail="Only scoring_failed applications can be retried"
        )

    asyncio.create_task(_trigger_ai_scoring(app_id))
    return application

    if (error.response?.status === 401) {

import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import type { User } from '@/types'

interface AuthState {
  user: User | null
  token: string | null
  setAuth: (user: User, token: string) => void
  clearAuth: () => void
  isAuthenticated: () => boolean
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      token: null,

      setAuth: (user, token) => {
        localStorage.setItem('access_token', token)
        set({ user, token })
      },

      clearAuth: () => {
        localStorage.removeItem('access_token')
        set({ user: null, token: null })
      },

      isAuthenticated: () => !!get().token,
    }),
    {
      name: 'auth-storage',
      partialize: (state) => ({ user: state.user, token: state.token }),
    }
  )
)

              rowKey="id"
              pagination={false}
              size="small"
            />
          </Card>
        </Col>
      </Row>
    </div>
  )
}

import { useParams, useNavigate } from 'react-router-dom'
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'
import {
  Card, Descriptions, Tag, Button, Select, Input,
  Modal, message, Row, Col, Progress, Steps, Spin,
} from 'antd'
import {
  RadarChart, Radar, PolarGrid, PolarAngleAxis,
  ResponsiveContainer, Tooltip,
} from 'recharts'
import { useState } from 'react'
import { applicationsApi } from '@/services/applications'
import { STATUS_CONFIG } from '@/constants/status'
import { VALID_TRANSITIONS_FRONTEND } from '@/constants/transitions'
import type { ApplicationStatus } from '@/types'
import { useAuthStore } from '@/store/authStore'

export default function ApplicationDetailPage() {
  const { id } = useParams<{ id: string }>()
  const navigate = useNavigate()
  const queryClient = useQueryClient()
  const user = useAuthStore((s) => s.user)
  const [modalOpen, setModalOpen] = useState(false)
  const [nextStatus, setNextStatus] = useState<ApplicationStatus | null>(null)
  const [notes, setNotes] = useState('')

  const { data: app, isLoading } = useQuery({
    queryKey: ['application', id],
    queryFn: () => applicationsApi.get(Number(id)),
    refetchInterval: (query) => {
      // AI 打分中时每 3 秒轮询一次
      const status = query.state.data?.status
      return status === 'ai_scoring' || status === 'submitted' ? 3000 : false
    },
  })

  const updateMutation = useMutation({
    mutationFn: ({ status, notes }: { status: ApplicationStatus; notes: string }) =>
      applicationsApi.updateStatus(Number(id), status, notes),
    onSuccess: () => {
      message.success('状态已更新')
      queryClient.invalidateQueries({ queryKey: ['application', id] })
      setModalOpen(false)
      setNotes('')
    },
    onError: (err: any) => {
      message.error(err.response?.data?.detail ?? '操作失败')
    },
  })

  if (isLoading) return <Spin size="large" style={{ display: 'block', marginTop: 80 }} />
  if (!app) return <div>申请不存在</div>

  const cfg = STATUS_CONFIG[app.status]
  const allowedNext = VALID_TRANSITIONS_FRONTEND[app.status] ?? []

  // AI 5维度雷达图数据
  const radarData = [
    { subject: '学历匹配', value: app.score_education ?? 0, full: 20 },
    { subject: '经验匹配', value: app.score_experience ?? 0, full: 25 },
    { subject: '技能匹配', value: app.score_skill ?? 0, full: 30 },
    { subject: '稳定性',   value: app.score_stability ?? 0, full: 15 },
    { subject: '整体印象', value: app.score_impression ?? 0, full: 10 },
  ].map((d) => ({ ...d, pct: (d.value / d.full) * 100 }))

  // 状态步骤条（主流程，不含 scoring_failed）
  const mainSteps = [
    'submitted', 'ai_scoring', 'ai_scored',
    'hr_review', 'dept_review', 'interview', 'offered', 'hired',
  ]
  const currentStep = mainSteps.indexOf(app.status)

  return (
    <div>
      <Button onClick={() => navigate(-1)} style={{ marginBottom: 16 }}>← 返回</Button>

      {/* 状态步骤条 */}
      <Card style={{ marginBottom: 16 }}>
        <Steps
          current={currentStep >= 0 ? currentStep : 0}
          status={app.status === 'rejected' ? 'error' : app.status === 'scoring_failed' ? 'error' : 'process'}
          size="small"
          items={mainSteps.map((s) => ({ title: STATUS_CONFIG[s as ApplicationStatus].label }))}
        />
      </Card>

      <Row gutter={16}>
        {/* 基本信息 */}
        <Col span={14}>
          <Card title="申请信息" style={{ marginBottom: 16 }}>
            <Descriptions column={2} size="small">
              <Descriptions.Item label="申请ID">{app.id}</Descriptions.Item>
              <Descriptions.Item label="岗位ID">{app.job_id}</Descriptions.Item>
              <Descriptions.Item label="候选人ID">{app.candidate_id}</Descriptions.Item>
              <Descriptions.Item label="简历ID">{app.resume_id}</Descriptions.Item>
              <Descriptions.Item label="当前状态">
                <Tag color={cfg.color}>{cfg.label}</Tag>
              </Descriptions.Item>
              <Descriptions.Item label="投递时间">{app.applied_at?.slice(0, 10)}</Descriptions.Item>
              {app.hr_notes && (
                <Descriptions.Item label="HR备注" span={2}>{app.hr_notes}</Descriptions.Item>
              )}
              {app.dept_notes && (
                <Descriptions.Item label="部门备注" span={2}>{app.dept_notes}</Descriptions.Item>

                </Descriptions.Item>
              )}
            </Descriptions>

            {/* 流转操作按钮 */}
            {allowedNext.length > 0 && (
              <div style={{ marginTop: 16 }}>
                <Button
                  type="primary"
                  onClick={() => setModalOpen(true)}
                >
                  推进状态
                </Button>
              </div>
            )}
          </Card>

          {/* AI 总结 */}
          {app.ai_summary && (
            <Card title="AI 分析总结" style={{ marginBottom: 16 }}>
              <p style={{ margin: 0 }}>{app.ai_summary}</p>
              <Row gutter={16} style={{ marginTop: 16 }}>
                <Col span={12}>
                  <div style={{ fontWeight: 600, marginBottom: 8, color: '#52c41a' }}>优势</div>
                  {(JSON.parse(app.ai_strengths ?? '[]') as string[]).map((s, i) => (
                    <div key={i} style={{ marginBottom: 4 }}>✅ {s}</div>
                  ))}
                </Col>
                <Col span={12}>
                  <div style={{ fontWeight: 600, marginBottom: 8, color: '#ff4d4f' }}>不足</div>
                  {(JSON.parse(app.ai_weaknesses ?? '[]') as string[]).map((s, i) => (
                    <div key={i} style={{ marginBottom: 4 }}>⚠️ {s}</div>
                  ))}
                </Col>
              </Row>
            </Card>
          )}
        </Col>

        {/* AI 打分面板 */}
        <Col span={10}>
          <Card title="AI 打分详情">
            {app.ai_total_score != null ? (
              <>
                <div style={{ textAlign: 'center', marginBottom: 16 }}>
                  <div style={{ fontSize: 48, fontWeight: 700, color: app.ai_total_score >= 75 ? '#52c41a' : app.ai_total_score >= 50 ? '#faad14' : '#ff4d4f' }}>
                    {app.ai_total_score.toFixed(1)}
                  </div>
                  <div style={{ color: '#888' }}>综合得分 / 100</div>
                </div>

                {radarData.map((d) => (
                  <div key={d.subject} style={{ marginBottom: 10 }}>
                    <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: 4 }}>
                      <span>{d.subject}</span>
                      <span>{d.value.toFixed(1)} / {d.full}</span>
                    </div>
                    <Progress
                      percent={Math.round(d.pct)}
                      size="small"

                      showInfo={false}
                    />

                ))}

                <ResponsiveContainer width="100%" height={220} style={{ marginTop: 16 }}>
                  <RadarChart data={radarData}>
                    <PolarGrid />

                    <Radar dataKey="pct" stroke="#667eea" fill="#667eea" fillOpacity={0.3} />

                ) : (
                  '暂无打分数据'
                )}

      </Row>

      <Modal
        title="推进申请状态"
        open={modalOpen}
        onOk={() => nextStatus && updateMutation.mutate({ status: nextStatus, notes })}

            rows={3}

## 第四步：完整 `backend/app/services/ai_service.py`

````python
import json
import re
import logging
from pathlib import Path
from typing import Optional
from dataclasses import dataclass

from openai import OpenAI
from app.core.config import settings

logger = logging.getLogger(__name__)


# ── 打分结果数据结构 ──────────────────────────────────────
@dataclass
class ScoreResult:
    education_match: float    # 满分 20
    experience_match: float   # 满分 25
    skill_match: float        # 满分 30
    stability: float          # 满分 15
    overall_impression: float # 满分 10
    total: float              # 满分 100
    summary: str
    strengths: list[str]
    weaknesses: list[str]


# ── 默认兜底分（打分失败时使用） ──────────────────────────
DEFAULT_SCORE = ScoreResult(
    education_match=0,
    experience_match=0,
    skill_match=0,
    stability=0,
    overall_impression=0,
    total=0,
    summary="AI打分失败，请人工审核",
    strengths=[],
    weaknesses=["AI自动打分未能完成，需人工评估"],
)


class AIService:
    def __init__(self):
        self.client = OpenAI(
            base_url=settings.LLM_BASE_URL,
            api_key=settings.LLM_API_KEY,
            timeout=settings.LLM_TIMEOUT,
        )
        self.model = settings.LLM_MODEL

    # ── 简历文本提取 ──────────────────────────────────────
    async def parse_resume_text(self, file_path: str) -> str:
        """从文件路径提取纯文本，支持 PDF / DOCX / TXT"""
        path = Path(file_path)
        if not path.exists():
            raise FileNotFoundError(f"简历文件不存在: {file_path}")

        suffix = path.suffix.lower()

        try:
            if suffix == ".pdf":
                return self._extract_pdf(path)
            elif suffix in (".docx", ".doc"):
                return self._extract_docx(path)
            elif suffix == ".txt":
                return path.read_text(encoding="utf-8", errors="ignore")
            else:
                raise ValueError(f"不支持的文件格式: {suffix}")
        except Exception as e:
            logger.error(f"简历文本提取失败: {e}")
            raise

    def _extract_pdf(self, path: Path) -> str:
        import pdfplumber
        texts = []
        with pdfplumber.open(path) as pdf:
            for page in pdf.pages:
                text = page.extract_text()
                if text:
                    texts.append(text)
        return "\n".join(texts)

    def _extract_docx(self, path: Path) -> str:
        from docx import Document
        doc = Document(str(path))
        return "\n".join(p.text for p in doc.paragraphs if p.text.strip())

    # ── 去除 Qwen3 的 <think>...</think> 思考标签 ────────
    def _strip_thinking(self, text: str) -> str:
        """Qwen3 会输出 <think>...</think> 思考过程，需要剥离"""
        return re.sub(r"<think>.*?</think>", "", text, flags=re.DOTALL).strip()

    # ── 从响应中提取 JSON ─────────────────────────────────
    def _extract_json(self, text: str) -> dict:
        """
        尝试从模型输出中提取 JSON，支持三种情况：
        1. 纯 JSON
        2. ```json ... ``` 代码块包裹
        3. JSON 混在其他文字中
        """
        # 先去掉思考标签
        text = self._strip_thinking(text)

        # 尝试直接解析
        try:
            return json.loads(text)
        except json.JSONDecodeError:
            pass

        # 尝试提取 ```json 代码块
        match = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", text, re.DOTALL)
        if match:
            try:
                return json.loads(match.group(1))
            except json.JSONDecodeError:
                pass

        # 尝试提取第一个完整的 { } 块
        match = re.search(r"\{.*\}", text, re.DOTALL)
        if match:

            elif suffix == ".txt":

        texts = []
        with pdfplumber.open(path) as pdf:
            for page in pdf.pages:
                text = page.extract_text()
                if text:
                    texts.append(text)
        return "\n".join(texts)

        from docx import Document
        doc = Document(str(path))
